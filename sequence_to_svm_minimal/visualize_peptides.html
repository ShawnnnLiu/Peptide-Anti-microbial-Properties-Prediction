def visualize_structure(pdb_path, title="", width=700, height=500, color='spectrum'):
    """Visualize a single PDB structure"""
    with open(pdb_path, 'r') as f:
        pdb_content = f.read()
    
    viewer = py3Dmol.view(width=width, height=height)
    viewer.addModel(pdb_content, 'pdb')
    
    if color == 'spectrum':
        viewer.setStyle({'cartoon': {'color': 'spectrum'}})
    elif color == 'plddt':
        viewer.setStyle({'cartoon': {'colorscheme': {'prop': 'b', 'gradient': 'roygb', 'min': 0, 'max': 1}}})
    else:
        viewer.setStyle({'cartoon': {'color': color}})
    
    viewer.setBackgroundColor('white')
    viewer.zoomTo()
    
    if title:
        print(f"\n {title}")
        print(f"   File: {pdb_path.name}")
    
    return viewer.show()

    amp_file = amp_dir / f"AMP_{amp_idx}.pdb"
    decoy_file = decoy_dir / f"DECOY_{decoy_idx}.pdb"
    
    with open(amp_file, 'r') as f:

def extract_plddt(pdb_path):
    """Extract pLDDT scores from PDB file"""
    parser = PDBParser(QUIET=True)
    structure = parser.get_structure('protein', pdb_path)
    model = structure[0]
    
    plddt_scores = []
    for chain in model:
        for residue in chain:
            for atom in residue:
                if atom.get_name() == 'CA':
                    plddt_scores.append(atom.get_bfactor())
    
    return plddt_scores

def show_gallery(files, start_idx=0, count=10, color='spectrum'):
    """Show a gallery of structures"""
    for i, pdb_file in enumerate(files[start_idx:start_idx+count], start_idx+1):
        # Get sequence info
        row = df[df['unique_id'] == pdb_file.stem]
        if len(row) > 0:
            seq = row['sequence'].values[0]
            label = "AMP" if row['label'].values[0] == 1 else "DECOY"
        else:
            seq = "N/A"
            label = "Unknown"
        
        print(f"\n{'='*60}")
        print(f" {pdb_file.stem} ({label})")
        print(f"   Sequence ({len(seq)} aa): {seq[:50]}{'...' if len(seq) > 50 else ''}")
        
        with open(pdb_file, 'r') as f:
            pdb_content = f.read()
        
        viewer = py3Dmol.view(width=500, height=350)
        viewer.addModel(pdb_content, 'pdb')
        if color == 'plddt':
            viewer.setStyle({'cartoon': {'colorscheme': {'prop': 'b', 'gradient': 'roygb', 'min': 0, 'max': 1}}})
        else:
            viewer.setStyle({'cartoon': {'color': color}})
        viewer.setBackgroundColor('white')
        viewer.zoomTo()
        viewer.show()

def find_and_visualize(search_term, max_results=5):
    """Find structures containing a sequence pattern"""
    matches = df[df['sequence'].str.contains(search_term.upper(), na=False)]
    
    if len(matches) == 0:
        print(f" No structures found containing '{search_term}'")
        return
    
    print(f" Found {len(matches)} structures containing '{search_term}'")
    print(f"   Showing first {min(max_results, len(matches))}:\n")
    
    for _, row in matches.head(max_results).iterrows():
        label_str = "AMP" if row['label'] == 1 else "DECOY"
        color = "green" if row['label'] == 1 else "red"
        
        print(f"\n{'='*60}")
        print(f" {row['unique_id']} ({label_str})")
        print(f"   {row['sequence']}")
        
        if row['label'] == 1:
            pdb_path = amp_dir / f"{row['unique_id']}.pdb"
        else:
            pdb_path = decoy_dir / f"{row['unique_id']}.pdb"
        
        with open(pdb_path, 'r') as f:
            pdb_content = f.read()
        
        viewer = py3Dmol.view(width=500, height=350)
        viewer.addModel(pdb_content, 'pdb')
        viewer.setStyle({'cartoon': {'color': color}})
        viewer.setBackgroundColor('white')
        viewer.zoomTo()
        viewer.show()

def view_by_id(structure_id):
    """View a specific structure by its ID (e.g., 'AMP_42' or 'DECOY_100')"""
    row = df[df['unique_id'] == structure_id]
    
    if len(row) == 0:
        print(f" Structure '{structure_id}' not found")
        print("   Valid formats: 'AMP_1' to 'AMP_286', 'DECOY_1' to 'DECOY_286'")
        return
    
    row = row.iloc[0]
    label_str = "Antimicrobial (AMP)" if row['label'] == 1 else "Non-antimicrobial (DECOY)"
    color = "green" if row['label'] == 1 else "red"
    
    print(f" {structure_id}")
    print(f"   Class: {label_str}")
    print(f"   Length: {row['length']} amino acids")
    print(f"   Sequence: {row['sequence']}")
    
    if row['label'] == 1:
        pdb_path = amp_dir / f"{row['unique_id']}.pdb"
    else:
        pdb_path = decoy_dir / f"{row['unique_id']}.pdb"
    
    # Show with pLDDT coloring
    with open(pdb_path, 'r') as f:
        pdb_content = f.read()
    
    print(f"\n Colored by pLDDT confidence (blue=high, red=low):")
    viewer = py3Dmol.view(width=700, height=500)
    viewer.addModel(pdb_content, 'pdb')
    viewer.setStyle({'cartoon': {'colorscheme': {'prop': 'b', 'gradient': 'roygb', 'min': 0, 'max': 1}}})
    viewer.setBackgroundColor('white')
    viewer.zoomTo()
    viewer.show()
